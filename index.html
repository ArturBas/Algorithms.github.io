Wirth's algorithm 1

September the 8th 2020

function searchingForValue (arrayLength, value) {

var array = []; var i = 0; var j = 0; var found; for (i; i < arrayLength; i++) { array.push(Math.floor(Math.random() * 10)); } console.log(array); for (j; j < arrayLength; j++) { if (array[j] == value) { console.log("The smallest component's index with value " + value + " is " + j); found = true; break; } } if (!found) { console.log("The array doesn't contain value " + value + " in any component"); } }; // Calling the 'searchingForValue'function: searchingForValue(30, 2); // put your own integer values for constants 'arrayLength' and 'value' into the function

function searchingForValue (arrayLength, value) { let array = []; let i = 0; let j = 0; for (i; i < arrayLength; i++) { array.push(Math.floor(Math.random() * 10)); } console.log(array); for (j; j < arrayLength; j++) { if (array[j] == value) { console.log("The smallest component's index with value " + value + " is " + j); break; } } if (j == arrayLength) { console.log("The array doesn't contain value " + value + " in any component"); } }; // Calling the 'searchingForValue'function: searchingForValue(4, 2); // put your own integer values for constants 'arrayLength' and 'value' into the function
