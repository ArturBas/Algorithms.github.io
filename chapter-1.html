Computing y = xⁿ, where n is a non-negative integer

Go to menu

Pascal (notation from ⇒ the book)



y:=1.0; i:=n;

WHILE i>0: DO (* x₀ⁿ = xⁱ * y *)

IF ODD(i) THEN y:=y*x END;

x:=x*x; i:=i DIV 2

END

JavaScript



let x;

let y;

const n;

let i;

let const isOdd = num => num % 2 === 1;


for (i = n; i > 0; i = Math.floor(i / 2)) {

if (isOdd(i)) {

y = y * x;

}

x = x * x;

}

JavaScript



let a = [];

let n = a.length;

let i = 1;

let k = 0;

let max = a[0];


for (i; i <= (n − 1); i++) {

if (max < a[i]) {

k = i;

max = a[k];

}

}

Dividing arrays (practical example)

Go to menu

Pascal (notation from ⇒ the book)



PROCEDURE Power (VAR W: Texts.Writer; N: INTEGER);

(*compute decimal representation of negative powers of 2*)

VAR i,k,r: INTEGER;

d: ARRAY N OF INTEGER;

BEGIN

FOR k:=0 TO N−1 DO

Texts.Write (W, "."); r: = 0;

FOR i:=1; TO k−1 DO

r:=10*r + d[i]; d[i]:=r DIV 2; r:=r MOD 2;

Texts.Write(W, CHR(d[i] + ord("0")))

END;

d[k]:=5; Texts.Writeln(W, "5"); Texts.WriteLn(W)

END

END Power

JavaScript



function power(n) {

let i;

let k;

let r;

let d = [];


for (k = 0; k <= (n − 1); k++) {

r = 0;

for (i = 0; i <= (k − 1); i++) {

r = 10 * r + d[i];

d[i] + Math.floor(r / 2);

r = r − 2 * d[i];

}

d[k] = 5;

console.log("." + d.join(""));

}

}
