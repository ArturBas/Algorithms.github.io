Finding the maximal element of an array

Pascal (⇒ notation from the book)
JavaScript

VAR a: ARRAY N OF INTEGER;

k:=0; max:=a[0];

FOR i:=1 TO N−1 DO

IF max < a[i] THEN k:=i; max:=a[k] END

END

const a = [];

const n = a.length;

let i = 1;

let k = 0;

let max = a[0];


for (i; i <= (n − 1); i++) {

if (max < a[i]) {

k = i;

max = a[k];

}

}

PROCEDURE Power (VAR W: Texts.Writer; N: INTEGER);

(*compute decimal representation of negative powers of 2*)

VAR i,k,r: INTEGER;

d: ARRAY N OF INTEGER;

BEGIN

FOR k:=0 TO N−1 DO

Texts.Write (W, "."); r: = 0;

FOR i:=1; TO k−1 DO

r:=10*r + d[i]; d[i]:=r DIV 2; r:=r MOD 2;

Texts.Write(W, CHR(d[i] + ord("0")))

END;

d[k]:=5; Texts.Writeln(W, "5"); Texts.WriteLn(W)

END

END Power

function power(n) {

let r;

let d = [];

r = 10 * r + d[i];

d[i] + Math.floor(r / 2);

}

}
